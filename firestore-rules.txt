// Firestore Security Rules for the fitness website
// These rules should be applied in the Firebase Console -> Firestore Database -> Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read/write access to leads collection for authenticated and anonymous users
    match /leads/{leadId} {
      allow read, write: if true; // Temporarily allow all access for testing
    }
    
    // Allow read/write access to users collection for authenticated users
    match /users/{userId} {
      allow read, write: if request.auth != null;
    }
    
    // Default deny all other documents
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

// Note: The above rules are permissive for testing purposes.
// In production, you should implement proper authentication checks:
// - Verify user identity
// - Limit write access to user's own data
// - Add rate limiting
// - Validate data structure
